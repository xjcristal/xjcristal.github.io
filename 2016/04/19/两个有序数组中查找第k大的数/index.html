两个有序数组中查找第k大的数

发表于 2016-04-19   |  

原文链接为算法剖析，解析得很通俗易懂，mark一下。具体解析看原文就很清晰明了，我在这里就贴上代码了。（其实也是摘录自原文，但是自己手动敲了一遍）。不得不感慨使用指针来截取数组还是很方便的。

int findKthElement(int *A, int m, int *B, int n, int k)
{
    if (m == 0) {
        return B[k-1];
    }
    if (n == 0) {
        return A[k-1];
    }
    int i = m >> 1, j = n >> 1, *p, *q;
    if (A[i] <= B[j]) {
        p = A;
        q = B;
    }else p = B, q = A, swap(i, j), swap(m, n);

    if (k <= i + j + 1) {
        return findKthElement(p, m, q, j, k);
    }else
        return findKthElement(p+i+1, m-i-1, q, n, k-i-1);
}

找中位数只是其中的一个特例，代码如下：

int findMediumFromTwoSortedArray(int *A, int m, int *B, int n) {
    if ( (m + n) % 2 == 0) {
        int k1 = (m + n) >> 1, k2 = k1 + 1;
        int mk1 = findKthElement(A, m, B, n, k1);
        int mk2 = findKthElement(A, m, B, n, k2);
        int m = (mk1 + mk2) >> 1;
        cout << "中位数为: "<< mk1 << "和 "<< mk2 <<"的平均数= "<< m <<endl;
        return m;
    }else {
        int k = (m + n) >> 1;
        int mk = findKthElement(A, m, B, n, k);
        cout << "中位数为："<< mk << endl;
        return mk;
    }
}

int main(int argc, const char * argv[]) {

    //生成的测试用例为array中的数，array＝｛1，2...maxN｝;
    int maxN = 12;
    vector<int> array;
    for (int i = 1; i <= maxN; i++) {
        array.push_back(i);
    }

    int flag = 100;//生成100次测试用例
    while (flag--) {
        int n1 = rand() % maxN;
        while (n1 == 0) {
            n1 = rand() % maxN;
        }

        int n2 = maxN - n1;

        int *array1 = new int[n1];
        map<int,int> tmap;
        for (int i = 0; i < n1; i++) {
            int indx = rand() % maxN;
            while (tmap.find(indx) != tmap.end()) {
                indx = rand() % maxN;
            }
            tmap.insert(pair<int, int>(indx,1));
            array1[i] = array[indx];
        }


        int *array2 = new int[n2];
        for (int i = 0; i < n2; i++) {
            int indx = rand() % maxN;
            while (tmap.find(indx) != tmap.end()) {
                indx = rand() % maxN;
            }
            tmap.insert(pair<int, int>(indx,1));
            array2[i] = array[indx];
        }

        sort(array1, array1+n1);
        sort(array2, array2+n2);

        // 到这一步，两个有序的测试数组已经生成。

        cout << "array1:{ " ;
        for (int i = 0; i < n1; i++) {
            cout << array1[i];
            if (i != n1-1) {
                cout << ",";
            }
        }
        cout << "}  ";

        cout << "array2:{ " ;
        for (int i = 0; i < n2; i++) {
            cout << array2[i];
            if (i != n2-1) {
                cout << ",";
            }
        }
        cout << "}  ";

        //求中位数
        int mthElement = findMediumFromTwoSortedArray(array1, n1, array2, n2);
        delete[] array1;
        delete[] array2;
    }
    return 0;
}
